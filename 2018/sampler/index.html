def is_odd(number) -> bool:
"""
	is_odd Module.
	Tests if the number passed to it is odd.
		params: number a numeric input.
	returns: Boolean
"""
	if type(number).__name__ == "int":
		return True if number & 1 else False
	elseif type(number).__name == "float":
		return True if number % 2 else False
	else:
		return False
